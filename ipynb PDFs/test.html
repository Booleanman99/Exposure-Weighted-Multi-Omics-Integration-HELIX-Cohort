import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler

# === Step 1: Load Data ===
methylome_child = pd.read_csv("clean_methylome_childhood.csv")
methylome_pregnancy = pd.read_csv("clean_methylome_pregnancy.csv")
base = pd.read_csv("../Helix_extracted.csv")

# === Step 2: Filter methylome data (non-zero, non-null effects + valid p-values) ===
methylome_child = methylome_child[
    (methylome_child['Effect'].notna()) & (methylome_child['Effect'] != 0) &
    (methylome_child['P.Value'].notna()) & (methylome_child['P.Value'] > 0)
]

methylome_pregnancy = methylome_pregnancy[
    (methylome_pregnancy['Effect'].notna()) & (methylome_pregnancy['Effect'] != 0) &
    (methylome_pregnancy['P.Value'].notna()) & (methylome_pregnancy['P.Value'] > 0)
]

# === Step 3: Normalize numeric exposure columns ===
base_z = base.copy()
numeric_cols = base_z.drop(columns=["ID"]).select_dtypes(include=[np.number]).columns
non_numeric_cols = base_z.drop(columns=["ID"]).select_dtypes(exclude=[np.number]).columns

print(" Numeric columns to normalize:", len(numeric_cols))
print(" Non-numeric columns skipped:", list(non_numeric_cols))

scaler = StandardScaler()
base_z[numeric_cols] = scaler.fit_transform(base_z[numeric_cols])

# === Step 4: Manual exposure name mapping ===
exposure_map = {
    'Indoor PMabsorbance': 'h_Absorbance_Log',
    'Indoor benzene': 'h_Benzene_Log',

# === Step 5: Risk scoring function ===
def compute_weighted_risk(df, methylome_df, risk_col):
    df[risk_col] = 0.0
    for _, row in methylome_df.iterrows():
        exposure = row["Exposure"]
        effect = row["Effect"]
        pval = row["P.Value"]
        weight = effect * (-np.log10(pval))

        if exposure in exposure_map:
            col = exposure_map[exposure]
            if col in df.columns:
                try:
                    df[risk_col] += df[col].fillna(0).astype(float) * weight
                except Exception as e:
                    print(f" Error with column {col}: {e}")

# === Step 6: Apply to both time points ===
compute_weighted_risk(base_z, methylome_child, "methylome_child_risk")
compute_weighted_risk(base_z, methylome_pregnancy, "methylome_pregnancy_risk")

# === Step 7: Compute total methylome risk ===
base_z["total_methylome_risk"] = base_z["methylome_child_risk"] + base_z["methylome_pregnancy_risk"]

# === Step 8: Save all three risk scores ===
output = base_z[["ID", "methylome_child_risk", "methylome_pregnancy_risk", "total_methylome_risk"]]

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from sklearn.model_selection import train_test_split

# === Step 1: Load risk score data ===
df = pd.read_csv("RISK_weighted.csv")

# === Step 2: Assign categorical risk tiers ===

    if score < 35:
        return 0  # Minimal Risk
    elif score < 65:
        return 1  # Moderate Risk
    else:
        return 2  # High Risk

df["risk_tier"] = df["risk_score"].apply(assign_risk_tier)

# === Step 3: Define input features and target ===
X = df.drop(columns=["ID", "risk_score", "risk_tier"])

)

model = XGBClassifier(
    objective="multi:softprob",
    num_class=3,
    eval_metric="mlogloss",
    use_label_encoder=False,
    random_state=42
)

model.fit(X_train, y_train)

# === Step 6: Predict and evaluate ===
y_pred = model.predict(X_test)

# Classification report
